## Basic Usage

```typescript
import { css } from '@sylphx/silk';

const button = css({
  display: 'flex',
  alignItems: 'center',
  padding: '0.75rem 1.5rem',
  backgroundColor: 'var(--color-primary)',
  color: 'white',
  borderRadius: '0.5rem'
});

<button className={button}>Click me</button>
```

## Configuration

Create a `silk.config.ts`:

```typescript
import { defineConfig } from '@sylphx/silk';

export default defineConfig({
  theme: {
    tokens: {
      colors: {
        primary: '#0070f3',
        secondary: '#8b5cf6'
      },
      spacing: {
        1: '0.25rem',
        2: '0.5rem',
        3: '0.75rem',
        4: '1rem'
      }
    }
  }
});
```

## Modern CSS Features

### Container Queries

```typescript
const card = css({
  containerType: 'inline-size',
  display: 'flex',
  flexDirection: 'column',

  '@container (min-width: 400px)': {
    flexDirection: 'row',
    gap: '1rem'
  }
});
```

### Modern Color Functions

```typescript
const themed = css({
  backgroundColor: 'oklch(0.7 0.15 180)',
  color: 'color-mix(in oklch, var(--color-primary) 80%, white)'
});
```

## React Integration

```typescript
import { css } from '@sylphx/silk';

function Button({ children, variant = 'primary' }) {
  const buttonClass = css({
    padding: '0.75rem 1.5rem',
    borderRadius: '0.5rem',
    ...(variant === 'primary' ? {
      backgroundColor: 'var(--color-primary)',
      color: 'white'
    } : {
      backgroundColor: 'transparent',
      border: '2px solid var(--color-primary)'
    })
  });

  return <button className={buttonClass}>{children}</button>;
}
```
